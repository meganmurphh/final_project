def avg_purch(column_names, df):
    size = 0
    costs = df["Cost"].tolist()
    total_costs = sum(costs)
    print("Total costs: ${0:5.2f}".format(total_costs))
    print("")

    purchases = df['Cost'].tolist()
    total_purch = len(purchases)
    print("Number of Purchases:",total_purch)
    print("")

    converted_costs = int(total_costs)
    converted_purch = int(total_purch)
    avg = converted_costs / converted_purch
    print("Average Costs: ${0:5.2f}".format(avg))

def max_min(df, column_names):
    df_numeric = df.set_index("Category").select_dtypes('number')
    idx = df_numeric.idxmax()
    maxcost = df["Cost"].max()
    idxmin = df_numeric.idxmin()
    mincost = df["Cost"].min()

    print("The most expensive purchase was ${0:.2f}".format(maxcost),"in",idx)
    print("The least expensive purchase was ${0:.2f}".format(mincost),"in",idxmin)

def main():
    column_names = ["Category", "Cost"]
    while True:
        try:
            file = input("Enter csv file to read: ")
            df = pd.read_csv(file, names=column_names)
            print("")
            avg_purch(column_names, df)
            print("")
            categories(column_names, df)
            max_min(df, column_names)
            print("")
            response = input("Would you like to read another file? (y/n) ")
            if (response == "y"):
                continue
            if (response == "n"):
                break
        except Exception as error:
            print(error)

main()
